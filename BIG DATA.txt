# Employee Record Storage and Retrieval Using HBase (Simulated)

class EmployeeDB:
    def __init__(self):
        # dictionary -> acts like HBase table
        self.table = {}

    def add_employee(self, emp_id, name, age, dept, email, salary):
        if emp_id in self.table:
            print(f"Employee ID {emp_id} already exists!")
        else:
            self.table[emp_id] = {
                "name": name,
                "age": age,
                "dept": dept,
                "email": email,
                "salary": salary
            }
            print("‚úÖ Employee added successfully!")

    def view_all(self):
        if not self.table:
            print("No records found.")
        else:
            print("\n--- Employee Records ---")
            for emp_id, data in self.table.items():
                print(f"ID: {emp_id}")
                for key, value in data.items():
                    print(f"   {key}: {value}")
                print("-----------------------")

    def search_employee(self, emp_id):
        emp = self.table.get(emp_id)
        if emp:
            print(f"\nEmployee ID: {emp_id}")
            for key, value in emp.items():
                print(f"   {key}: {value}")
        else:
            print("‚ùå Employee not found.")

    def delete_employee(self, emp_id):
        if emp_id in self.table:
            del self.table[emp_id]
            print("üóëÔ∏è Employee deleted successfully!")
        else:
            print("Employee ID not found.")

    def update_employee(self, emp_id, field, new_value):
        if emp_id in self.table:
            if field in self.table[emp_id]:
                self.table[emp_id][field] = new_value
                print("‚úÖ Employee record updated successfully!")
            else:
                print("Invalid field name.")
        else:
            print("Employee not found.")

# Main Program
def main():
    db = EmployeeDB()

    while True:
        print("\n--- Employee Record Storage and Retrieval ---")
        print("1. Add Employee")
        print("2. View All Employees")
        print("3. Search Employee")
        print("4. Delete Employee")
        print("5. Update Employee")
        print("6. Exit")

        choice = input("Enter your choice: ")

        if choice == '1':
            emp_id = input("Enter Employee ID: ")
            name = input("Enter Name: ")
            age = input("Enter Age: ")
            dept = input("Enter Department: ")
            email = input("Enter Email: ")
            salary = input("Enter Salary: ")
            db.add_employee(emp_id, name, age, dept, email, salary)

        elif choice == '2':
            db.view_all()

        elif choice == '3':
            emp_id = input("Enter Employee ID to search: ")
            db.search_employee(emp_id)

        elif choice == '4':
            emp_id = input("Enter Employee ID to delete: ")
            db.delete_employee(emp_id)

        elif choice == '5':
            emp_id = input("Enter Employee ID to update: ")
            field = input("Enter field to update (name/age/dept/email/salary): ")
            new_value = input("Enter new value: ")
            db.update_employee(emp_id, field, new_value)

        elif choice == '6':
            print("Exiting... Thank you!")
            break
        else:
            print("Invalid choice! Please try again.")

if __name__ == "__main__":
    main()
